Lambda Function 2 (S3 to DynamoDB)


import boto3
from uuid import uuid4
import urllib.parse

def lambda_handler(event, context):

    # Initialize DynamoDB
    dynamodb = boto3.resource('dynamodb')
    table = dynamodb.Table('newtable12')   # your table name

    if 'Records' in event:

        for record in event['Records']:

            # Extract S3 details
            bucket_name = record['s3']['bucket']['name']
            object_key = urllib.parse.unquote_plus(
                record['s3']['object']['key']
            )
            size = record['s3']['object'].get('size', -1)
            event_name = record.get('eventName', 'Unknown')
            event_time = record.get('eventTime', 'Unknown')

            # Ensure only your bucket is processed
            if bucket_name == "week71f":

                try:
                    table.put_item(
                        Item={
                            'unique': str(uuid4()),
                            'Bucket': bucket_name,
                            'Object': object_key,
                            'Size': size,
                            'Event': event_name,
                            'EventTime': event_time
                        }
                    )

                    print("SUCCESS: Data inserted into DynamoDB")

                except Exception as e:
                    print("ERROR:", str(e))

    else:
        print("No 'Records' key found in the event.")
